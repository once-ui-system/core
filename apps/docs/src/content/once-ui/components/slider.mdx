The Slider component provides an intuitive way to select a value from a range using a draggable thumb along a track.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  preview={
    <SliderBasicExample />
  }
  codes={[
    {
      code:
`const [value, setValue] = useState(50);

return (
    <Slider
      value={value}
      onChange={setValue}
      min={0}
      max={100}
    />
);`,
      language: "tsx",
      label: "Slider"
    }
  ]}
/>
