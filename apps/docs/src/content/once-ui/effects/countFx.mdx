CountFx is a wrapper for numbers with different animation effects.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  reloadButton
  preview={
    <CountFxExample />
  }
  codes={[
    {
      code:
`const [value, setValue] = useState<number>(0);

useEffect(() => {
    const targetValue = Math.floor(Math.random() * 9000) + 1000;
    setValue(targetValue);
}, []);

<CountFx
    variant="display-strong-m"
    value={value}
    speed={5000}
    effect="wheel"
    easing="ease-out"
/>

<CountFx
    variant="display-strong-m"
    value={value}
    speed={5000}
    effect="smooth"
    easing="ease-out"
/>

<CountFx
    variant="display-strong-m"
    value={value}
    speed={5000}
    separator=","
    effect="simple"
    easing="ease-out"
/>`,
      language: "tsx",
      label: "CountFx"
    }
  ]}
/>

## Decimals

CountFx supports decimal values with the `decimals` prop.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  reloadButton
  preview={
    <CountFxDecimalExample />
  }
  codes={[
    {
      code:
`const [value1, setValue1] = useState<number>(0);
const [value2, setValue2] = useState<number>(0);

useEffect(() => {
    const targetValue1 = parseFloat((Math.random() * 200 + 50).toFixed(2));
    const targetValue2 = parseFloat((Math.random() * 150 + 25).toFixed(2));

    setValue1(targetValue1);
    setValue2(targetValue2);
}, []);

<CountFx
    variant="display-strong-m"
    value={value1}
    decimals={2}
    speed={2000}
    effect="simple"
    easing="ease-out"
/>

<CountFx
    variant="display-strong-m"
    value={value2}
    decimals={2}
    speed={2000}
    separator=","
    effect="simple"
    easing="ease-out"
    children=" USD"
/>`,
      language: "tsx",
      label: "Decimals"
    }
  ]}
/>
