## Decimals

CountFx supports decimal values with the `decimals` prop.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  reloadButton
  preview={
    <CountFxDecimalExample />
  }
  codes={[
    {
      code:
`const [value1, setValue1] = useState<number>(0);
const [value2, setValue2] = useState<number>(0);

useEffect(() => {
    const targetValue1 = parseFloat((Math.random() * 200 + 50).toFixed(2));
    const targetValue2 = parseFloat((Math.random() * 150 + 25).toFixed(2));

    setValue1(targetValue1);
    setValue2(targetValue2);
}, []);

<CountFx
    variant="display-strong-m"
    value={value1}
    decimals={2}
    speed={2000}
    effect="simple"
    easing="ease-out"
/>

<CountFx
    variant="display-strong-m"
    value={value2}
    decimals={2}
    speed={2000}
    separator=","
    effect="simple"
    easing="ease-out"
    children=" USD"
/>`,
      language: "tsx",
      label: "Decimals"
    }
  ]}
/>

## Trigger on scroll

Use the [useInViewport](/once-ui/hooks/useInViewport) hook to trigger the count animation when the element enters the viewport. This example starts counting when the element reaches the center of the screen.

<CodeBlock
  marginTop="16"
  marginBottom="24"
  preview={
    <ScrollCountExample />
  }
  codes={[
    {
      code:
`import { useRef, useState, useEffect } from "react";
import { CountFx, useInViewport } from "@once-ui-system/core";

function ScrollCountExample() {
    const ref = useRef<HTMLDivElement>(null);
    const [value, setValue] = useState<number>(0);
    
    // Trigger when element reaches center of viewport
    // rootMargin: "-50% 0px" creates a line at the vertical center
    const isInViewport = useInViewport(ref, { 
        rootMargin: "-50% 0px" 
    });
    
    useEffect(() => {
        if (isInViewport) {
            setValue(9876);
        }
    }, [isInViewport]);
    
    return (
      <CountFx
        variant="display-strong-l"
        value={value}
        speed={2000}
        effect="wheel"
        easing="ease-out"
      />
    );
}`,
      language: "tsx",
      label: "Scroll trigger"
    }
  ]}
/>
